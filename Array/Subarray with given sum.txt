Solution-

#include <bits/stdc++.h>
using namespace std;

void findsubstring(int arr[],int n,int target){
    int start=0,end=0;
    int currentsum=arr[start];
    while(currentsum!=target){
        if(currentsum<target){
            end++;
            currentsum+=arr[end];
        }
        if(currentsum>target){
            currentsum-=arr[start];
            start++;
        }
        if(start>n || end>n){
            break;
        }
    }
    if(start>n || end>n){
        cout<<-1;
    }else{
        cout<<start+1<<" "<<end+1;
    }
}

int main() {
	int t;
	cin>>t;
	while(t--){
	    int n,target;
	    cin>>n>>target;
	    int arr[n];
	    for(int i =0;i<n;i++){
	        cin>>arr[i];
	    }
	    findsubstring(arr,n,target);
	    cout<<endl;
	}
	return 0;
}
