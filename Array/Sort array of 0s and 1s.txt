#include<bits/stdc++.h>
using namespace std;

void sortArray(int arr[], int n){
    int i=0,j=n-1;
    while(i<j){
        if(arr[i]==0){
            i++;
        }else{
            swap(arr[i],arr[j]);
            j--;
        }
    }
}

int main(){
    int n=9;
    int arr[n]={1,1,0,0,1,1,0,0,0};
    sortArray(arr,n);
    for(int i=0;i<n;i++){
        cout<<arr[i]<<" ";
    }
    cout<<endl;
    return 0;
}
