Solution:-

#include <bits/stdc++.h>
using namespace std;

bool solution(int arr[],int n){
    int maximum=0;
    for(int i=0;i<n;i++){
        maximum=max(arr[i],maximum);
    }
    int hash[maximum+1]={0};
    for (int i=0;i<n;i++){
        hash[arr[i]]++;
    }
    for (int i=1;i<maximum+1;i++){
        if (hash[i]==0)
            continue;
        for (int j = 1; j < maximum + 1; j++) {
            if(i==j && hash[i]==1 || hash[j]==0)
                continue;
            int val = i+j;
            if (val > maximum) 
                continue;
            if(hash[val])
                return true;
        }
    }
    return false;
    
}
int main(){
	int t;
	cin>>t;
	while(t--){
	    int size;
	    cin>>size;
	    int arr[size];
	    for(int i=0;i<size;i++){
	        cin>>arr[i];
	        arr[i]=arr[i]*arr[i];
	    }
	    if(solution(arr,size))
	        cout<<"Yes"<<endl;
	    else
	        cout<<"No"<<endl;
	}
	return 0;
}
