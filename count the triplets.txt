#include<bits/stdc++.h> 
using namespace std; 

long int countTriplets(int arr[], int n){
	sort(arr, arr + n); long int t=0;
	for (int i = n - 1; i >= 0; i--){
		int j = 0;
		int k = i - 1;
		while (j < k){
			if (arr[i] == arr[j] + arr[k])
			  {t++;j++;k--;}
			else if (arr[i] > arr[j] + arr[k])
				j++;
			else
				k--;
		}
	}
	if(t!=0)
	    return t;
	return -1;
    
}

int main() {
	int t;
	cin>>t;
	while(t--){
	    int n;
	    cin>>n;
	    int arr[n];
	    
	    for (int i=0;i<n;i++){
	        cin>>arr[i];
	    }
	    cout<<countTriplets(arr,n)<<endl;
	}
	return 0;
}
